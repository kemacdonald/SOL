# Specify the model for standardized data:
model {
    for ( i in 1:Ntotal ) {
        y[i] ~ dnorm( zbeta0 + zbeta1 * x[i] , 1/(zsigma)^2 )
    }
    # Prior for intercept is vague on standardized scale
    zbeta0 ~ dnorm(0,1/10) 
    # Prior for slope is truncated at zero and contsrained to contain plausible a priori slopes
    zbeta1 ~ dnorm(0,1)I(0,)
    # Prior on zsigma is vague on standardized scale
    zsigma ~ dunif( 1.0E-3 , 1.0E+3 )
    # Prior predictive
    zbeta1_prior ~ dnorm(0,1)I(0,)
    zbeta0_prior ~ dnorm(0,1/10) 
}
