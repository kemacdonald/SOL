# Specify the model for standardized data:
model {
        # Bayesian Linear Regression
        for ( i in 1:Ntotal ) {
            y[i] ~ dnorm( zbeta0 + zbeta1 * x[i] , 1/(zsigma)^2 )
        }
        # Priors 
        zbeta0 ~ dnorm(0,1/10) 
        zbeta1 ~ dnorm(0,1)I(,0)
        zsigma ~ dunif( 1.0E-3 , 1.0E+3 )
        # Prior predictive
        zbeta0_prior ~ dnorm(0,1/10) 
        zbeta1_prior ~ dnorm(0,1)I(,0)
}
